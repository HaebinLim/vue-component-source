      :theme="theme"
      :appearance="appearance"
      :label="label"
      :use-under-line="useUnderLine"
      :has-under-line="hasUnderLine"
      @click.prevent.native="log"
    />
  `,
  methods: {
    log(event) {
      if (this.to) return;
      action(`HTML Anchor clicked`)(this.href);
    }
  }
});

# &lt;TextLink> components

`to` 또는 `href` property에 의해 `<nuxt-link>` 또는 HTML native `<a>` 
엘리먼트로 렌더링되며 `<nuxt-link>`의 property, `<a>`의 attribute 적용 또한
이 값에 의해 결정 됨

둘 모두 적용 시에는 `to`가 우선권을 가짐.

아래는 앞에서부터 차례대로 기본, 밑줄 애니메이션 적용, 밑줄 애니메이션 및 초기 밑줄이 적용된 
사례
