      description: 'click event handler로 동작시킬 함수'
    }
  }}
/>

export const Template = (args, { argTypes }) => ({
  props: [...Object.keys(argTypes)],
  slots: {},
  components: {TextButton},
  template: `
    <TextButton
      :type="type"
      :theme="theme"
      :appearance="appearance"
      :label="label"
      :disabled="disabled"
      :use-under-line="useUnderLine"
      :has-under-line="hasUnderLine"
      :click-callback="clickCallback"
      @click="log"
    />
  `,
  methods: {
    log() {
      action(`button clicked`)();
    }
  }
});


# &lt;TextButton> components

아래는 앞에서부터 차례대로 
- 기본
- 밑줄 애니메이션 적용
- 밑줄 애니메이션 및 초기 밑줄
- 비활성
- 비활성 & 밑줄 애니메이션
- 비활성 & 밑줄 애니메이션 및 초기 밑줄
