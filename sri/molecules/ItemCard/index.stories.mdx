import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import { action } from '@storybook/addon-actions';
import ItemCard from './index';

<Meta
  title="Molecules/ItemCard"
  component={ItemCard}
  argTypes={{
    useOptionalContent: {
      description: `Optional 항목들의 노출 여부 정의`,
      table: {
        type: {
          summary: `Object`,
          detail: `{category, desc, editorName, checked, replies, likes}`,
        }
      }
    },
  }}
/>


export const Template = (args, { argTypes }) => ({
  props: [...Object.keys(argTypes)],
  components: {ItemCard},
  template: `
    <ItemCard
      :article-id="articleId"
      :title="title"
      :image="image"
      :archived="archived"
      :category="category"
      :list-category="listCategory"
      :desc="desc"
      :replies="replies"
      :likes="likes"
      :editor-name="editorName"
      :editor-id="editorId"
      :checked="checked"
      :appearance="appearance"
      :isEmphasis="isEmphasis"
      :isNewWindow="isNewWindow"
      :is-skeleton="isSkeleton"
      :use-optional-content="useOptionalContent"
      @change-selected="logClick"
      @change-archive="logChange"
    />
  `,
  mounted() {
    this.getDelayedListData();
  },  
  methods : {
    logClick(payload) {
      action(`click`)(payload);
    },
    logChange(payload) {
      action(`change`)(payload);
    },
    getDelayedListData() {
      return new Promise(resolve => {
        setTimeout(() => {
          this.isSkeleton = null;
          resolve();
        }, 1000);
      });
    }
  },
});
